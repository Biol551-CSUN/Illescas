Creating a Plot with Penguins Data

penguins %>%
  ggplot(aes(x = bill_length_mm, # this allows us to create a scatter plot grouped by island with different colors for each island to differentiate
          y = bill_depth_mm, 
         color = island,
         group = island)) +
  geom_point(shape = 17, size = 3) +
  labs(x = "Bill Length (mm)", y = "Bill Depth (mm)",
       title = "Bill Length and Depth According to Island",
       color = "Island") +  # this allows us to change the legend title
  theme_bw()+ # This gives us a cleaner graph with dotted lines
  theme(legend.position = "right", plot.title = element_text(size = 15, face = "bold"))

Create a Table with Penguins Data

penguins %>%
  group_by(island) %>% # this allows us to group by island
  select(bill_length_mm, body_mass_g) %>% # this allows us to select the bill length and the body mass columns
  summarise(billmean = mean(bill_length_mm, na.rm = TRUE), # calculates mean for bill length
            bodymean = mean(body_mass_g, na.rm = TRUE)) %>%# this allows us to calculate mean for body mass
  kbl() %>% # creates a kable table
  kable_styling(full_width = FALSE) %>% # this allows us to set the width to be thinner so it doesn't take up the whole page
  add_header_above(c("Island" = 1, "Bill Length Mean (mm)" = 1, "Body Mass Mean (g)" = 1)) # this allows us to add new column names to include units of measure.  For the life of me, I couldn't figure out how to remove the default column names :(
